pipeline {
    agent any

    environment {
        IMAGE_NAME = 'my-php-apache'
        CONTAINER_NAME = 'my-php-container'
    }

    stages {
        stage('Checkout') {
            steps {
                // This stage checks out the source code from your version control system (e.g., Git)
                script {
                    checkout scm
                }
            }
        }

        stage('Build and Deploy') {
            steps {
                script {
                    // Build the Docker image
                    docker.build IMAGE_NAME, '-f Dockerfile .'

                    // Run the Docker container
                    docker.image(IMAGE_NAME).withRun('-p 8080:80 --name ${CONTAINER_NAME}') {
                        // Container is running with Apache serving your PHP application
                        sh 'sleep infinity'
                    }
                }
            }
        }
    }

    post {
        always {
            // Clean up the container after the build
            script {
                docker.stop(CONTAINER_NAME)
                docker.removeContainer(CONTAINER_NAME)
            }
        }
    }
}
